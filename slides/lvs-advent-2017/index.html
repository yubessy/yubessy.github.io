from networkx import from_pandas_dataframe, spring_layout, draw
df = read_sql('''
    WITH tmp_users AS (
        SELECT
            likes.user_id
        FROM
            likes
            INNER JOIN users ON likes.user_id = users.id
        WHERE
            users.calendar_participation = 1
        GROUP BY
            likes.user_id
        HAVING
            COUNT(*) <= 10
    )
    SELECT
        likes.user_id AS from_user,
        items.user_id AS to_user
    FROM
        likes
        INNER JOIN items ON likes.item_id = items.id
    WHERE
        likes.user_id IN tmp_users
''', con)
graph = from_pandas_dataframe(df, 'from_user', 'to_user')
pos = spring_layout(graph, k=1.5)
draw(graph, pos=pos, with_labels=True, node_size=1500, alpha=0.5, arrows=True, )

from networkx import from_pandas_dataframe, DiGraph, pagerank
df = read_sql('''
    SELECT
        likes.user_id AS from_user,
        items.user_id AS to_user
    FROM
        likes
        INNER JOIN items ON likes.item_id = items.id
        INNER JOIN users ON likes.user_id = users.id
    WHERE
        users.calendar_participation = 1
''', con)
graph = from_pandas_dataframe(df, source='from_user', target='to_user', create_using=DiGraph())
ranking = pagerank(graph) # PageRank
df = DataFrame([{'user_id': k, 'score': v} for k, v in ranking.items()], columns=['user_id', 'score'])
df = df.sort_values(by='score', ascending=False)
df[:10].iloc[::-1].plot.barh(x='user_id', y='score', color='c', alpha=0.8, figsize=(8, 4))

# カレンダー取得
import re
from datetime import date

import requests
import lxml.html as html
from pandas import DataFrame

cal_urls = {
    "関": "https://qiita.com/advent-calendar/2017/livesense-kan",
    "学": "https://qiita.com/advent-calendar/2017/livesense-gaku",
    "自": "https://qiita.com/advent-calendar/2017/livesense-ji",
}

if input("Qiitaからカレンダーを取得しますか？ yes/no > ") == 'yes':
    items = []
    for calendar, url in cal_urls.items():
        print(url)
        body = requests.get(url).text
        dom = html.fromstring(body)
        for day_el in dom.xpath("//td[@class='adventCalendarCalendar_day']"):
            day = day_el.xpath("./p[@class='adventCalendarCalendar_date']/text()")[0]
            dt = date(2017, 12, int(day))
            author = day_el.xpath("./div[@class='adventCalendarCalendar_author']/a/@href")[0][1:]
            title = day_el.xpath("./div[@class='adventCalendarCalendar_comment']/a/text()")[0]
            url = day_el.xpath("./div[@class='adventCalendarCalendar_comment']/a/@href")[0]
            item_id = re.match(r".+/items/([0-9a-f]+)", url).group(1)
            item = dict(id=item_id, calendar=calendar, date=dt, user_id=author, title=title, url=url)
            items.append(item)    
    df = DataFrame(items, columns=['id', 'calendar', 'date', 'user_id', 'url', 'titlr'])
    df.to_csv('data/items.csv', index=None)

import json

from pathlib import Path

import requests
from pandas import read_csv

# いいね取得
import json
import os
from pathlib import Path

import requests
from pandas import read_csv

ENDPOINT = "https://qiita.com/api/v2"
TOKEN = os.environ['QIITA_ACCESS_TOKEN']
HEADERS = {"Authorization": f"Bearer {TOKEN}"}
MAX_PAGE_REQUEST_LIMIT = 100
PER_PAGE = 100

items_df = read_csv('data/items.csv')
items_dir = Path('data/items')
if input("Qiitaからいいね一覧を取得しますか？ yes/no > ") == 'yes':
    for i, row in items_df.iterrows():
        print(row['id'], row['calendar'], row['date'], row['user_id'])
        item_id = row['id']
        likes_dir = items_dir / item_id / 'likes'
        likes_dir.mkdir(parents=True, exist_ok=True)
        for page in range(1, MAX_PAGE_REQUEST_LIMIT + 1):
            params = {'page': page, 'per_page': PER_PAGE}
            res = requests.get(f"{ENDPOINT}/items/{item_id}/likes", headers=HEADERS, params=params)
            likes_json = json.dumps(json.loads(res.text, strict=False), ensure_ascii=False, indent=2)
            (likes_dir / f'{page:0>3d}.json').write_text(likes_json)

            total_count = int(res.headers['Total-Count'])
            if page * PER_PAGE >= total_count:
                break

# 記事インデックス
import json
import sqlite3
from pathlib import Path

from pandas import read_csv

items_df = read_csv('data/items.csv')
con = sqlite3.connect("data/dataset.db", isolation_level=None)
for _, row in items_df.iterrows():
    item_file = Path(f'data/items/{row["id"]}/item.json')
    item = json.loads(item_file.read_text())
    user = item["user"]
    con.execute('''
        INSERT OR REPLACE INTO users(
            id, calendar_participation, name, description, profile_image_url,
            items_count, followees_count, followers_count
        ) VALUES (
            ?, ?, ?, ?, ?,
            ?, ?, ?
        )
    ''', (
        user["id"], True, user["name"], user["description"], user["profile_image_url"],
        user["items_count"], user["followees_count"], user["followers_count"]
    ))
    con.execute('''
        INSERT OR REPLACE INTO items(
            id, user_id, url, title, likes_count, comments_count,
            calendar_name, calendar_date, created_at, updated_at
        ) VALUES (
            ?, ?, ?, ?, ?, ?,
            ?, ?, ?, ?
        )
    ''', (
        item["id"], user["id"], item["url"], item["title"], item["likes_count"], item["comments_count"],
        row["calendar"], row["date"], item["created_at"], item["updated_at"]
    ))
    con.execute('''
        INSERT OR REPLACE INTO item_bodies(
            item_id, body, rendered_body
        ) VALUES (
            ?, ?, ?
        )
    ''', (
        item["id"], item["body"], item["rendered_body"]
    ))
    con.executemany('''
        INSERT OR REPLACE INTO tags(
            item_id, name
        ) VALUES (
            ?, ?
        )
    ''', [
        (item["id"], tag['name']) for tag in item["tags"]
    ])

# いいねインデックス
import json
import sqlite3
from pathlib import Path

from pandas import read_csv

items_df = read_csv('data/items.csv')
con = sqlite3.connect("data/dataset.db", isolation_level=None)
for i, row in items_df.iterrows():
    item_id = row['id']
    for likes_file in Path(f'data/items/{item_id}/likes').glob('*.json'):
        likes = json.loads(likes_file.read_text())
        for like in likes:
            user = like['user']
            # 記事書いて無い人のみ
            if not con.execute('SELECT 1 FROM users WHERE id = ?', (user['id'],)).fetchone():
                con.execute('''
                    INSERT OR REPLACE INTO users(
                        id, calendar_participation, name, description, profile_image_url,
                        items_count, followees_count, followers_count
                    ) VALUES (
                        ?, ?, ?, ?, ?,
                        ?, ?, ?
                    )
                ''', (
                    user["id"], False, user["name"], user["description"], user["profile_image_url"],
                    user["items_count"], user["followees_count"], user["followers_count"]
                ))

            con.execute('''
                INSERT OR REPLACE INTO likes(
                    user_id, item_id, created_at
                ) VALUES (
                    ?, ?, ?
                )
            ''', (
                user["id"], row['id'], like['created_at']
            ))

# キーワード抽出
from pathlib import Path

import lxml.html as html
import sqlite3

from mecab_wrapper import Parser

parser = Parser('ipadic')

STOPS_JA = Path('stopword-ja.txt').read_text().strip().split()
STOPS_EN = Path('stopword-en.txt').read_text().strip().split()
POSS = ('名詞', '動詞', '形容詞', '形容動詞')

def is_keyword_node(node):
    return node.pos in POSS and node.base not in STOPS_JA and node.base not in STOPS_EN

con = sqlite3.connect("data/dataset.db", isolation_level=None)

for item_id, rendered_body in con.execute('SELECT item_id, rendered_body FROM item_bodies').fetchall():
    dom = html.fromstring(rendered_body)
    code_els = dom.xpath('//code|//div[@class="code-frame"]')
    for code_el in code_els:
        code_el.getparent().remove(code_el)
    text = dom.text_content()
    nodes = parser.parse(text)
    keywords = [n.base for n in nodes if is_keyword_node(n)]
    con.execute('UPDATE item_bodies SET keywords = ? WHERE item_id = ?', (' '.join(keywords), item_id))

# ワードクラウド用テキスト
from pathlib import Path
from pandas import read_sql

con = sqlite3.connect("data/dataset.db", isolation_level=None)
df = read_sql('''
    SELECT
        items.calendar_date,
        items.calendar_name,
        item_bodies.keywords
    FROM
        items
        INNER JOIN item_bodies
            ON items.id = item_bodies.item_id
''', con)

df = df.pivot(index='calendar_date', columns='calendar_name', values='keywords')
for col in df:
    keywords = ' '.join(df[col].tolist())
    Path(f'data/kw-{col}.txt').write_text(keywords)
